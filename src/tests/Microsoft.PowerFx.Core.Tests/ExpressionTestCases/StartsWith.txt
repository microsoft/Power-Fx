// ******** STRING PARAMETERS ********

// Basic positive case
>> StartsWith("Hello","H")
true

// Basic negative case
>> StartsWith("Hello","A")
false

// Basic positive multichar case
>> StartsWith("Hello","He")
true

// Basic negative multichar case
>> StartsWith("Hello","Pq")
false

// Substring case
>> StartsWith("Hello","ell")
false

// Complete string case
>> StartsWith("Hello","Hello")
true

// Extra letters case
>> StartsWith("Hello","Hello sir")
false

// Case sensitive case
>> StartsWith("Hello","hE")
true

// Empty text case
>> StartsWith("","Hi")
false

// Empty end case
>> StartsWith("Hello","")
true

// Empty both case
>> StartsWith("","")
true

// Space begins text case
>> StartsWith(" A B","A B")
false

>> StartsWith(" A B "," A B")
true

// Space around ends case
>> StartsWith("A B ","A ")
true

// One char longer end case
>> StartsWith("a","aa")
false

// ******** NUMBER PARAMETERS ********

// Basic positive case
>> StartsWith(123456,"1")
true

>> StartsWith(123456,1)
true

// Basic negative case
>> StartsWith(123456,"A")
false

>> StartsWith(123456,3)
false

>> StartsWith("Hello",1)
false

// Basic positive multichar case
>> StartsWith(123456,"12")
true

>> StartsWith(123456,12)
true

// Basic negative multichar case
>> StartsWith(123456,"abc")
false

>> StartsWith(123456,786)
false

>> StartsWith("Hello",14)
false

// Substring case
>> StartsWith(12356,356)
false

>> StartsWith(12356,"356")
false

// Complete string case
>> StartsWith(123456,"123456")
true

>> StartsWith(123456,123456)
true

// Extra letters case
>> StartsWith(123456,"123456 789")
false

// Empty text case
>> StartsWith("",123456)
false

// Empty end case
>> StartsWith(123456,"")
true

// One char longer end case
>> StartsWith(123456,1234567)
false

//Decimal numbers
>> StartsWith(0.123456,"0.")
true


// ******** NULL and ERROR PARAMETERS ********

// Invalid operation: Division by zero
>> StartsWith("hello",1/0)
Error(Kind=Div0)

// Invalid operation: Division by zero
>> StartsWith(1/0,"hello")
Error(Kind=Div0)

// Null text case - coerced to ""
>> StartsWith(Blank(),"ok")
false

// Null start case - coerced to ""
>> StartsWith("Hello",Blank())
true

// Null start case - coerced to ""
>> StartsWith("",Blank())
true

// Null both case - coerced to ""
>> StartsWith(Blank(),Blank())
true


// ******** BOOLEAN PARAMETERS ********

>> StartsWith(true,true)
true

>> StartsWith(false,false)
true

>> StartsWith(true,false)
false

>> StartsWith(false,true)
false

>> StartsWith(12345,true)
false

>> StartsWith("0Hello",false)
false

>> StartsWith("true","false")
false

>> StartsWith("false boolean value is coerced to the string false", false)
true

>> StartsWith("true boolean value is coerced to the string true", true)
true

// ******** FUNCTION VALUE AS PARAMETERS ********
>> StartsWith("HelloWorld",Left("HelloGoodMorning", 5))
true

>> StartsWith(54678,Len("Hello"))
true
