
>> ForAll(GetProperties(ParseJSON("{""a"":1,""b"":true}")),{Name:ThisRecord.Name,StrValue:Text(ThisRecord.Value)})
Table({Name:"a",StrValue:"1"},{Name:"b",StrValue:"true"})

>> ForAll(GetProperties(ParseJSON("{""a"":1,""b"":true,""c"":null,""d"":[1,2,3],""e"":""hello""}")),{Name:ThisRecord.Name})
Table({Name:"a"},{Name:"b"},{Name:"c"},{Name:"d"},{Name:"e"})

>> CountRows(GetProperties(ParseJSON("{""a"":1,""b"":true,""c"":null,""d"":[1,2,3],""e"":""hello""}")))
5

>> Concat(GetProperties(ParseJSON("{""a"":1,""b"":true,""c"":null,""e"":""hello""}")),$"{ThisRecord.Name}:{ThisRecord.Value}", ",")
"a:1,b:true,c:,e:hello"

>> CountRows(LookUp(GetProperties(ParseJSON("{""a"":1,""b"":true,""c"":null,""d"":[1,2,3],""e"":""hello""}")),Name="d",Value))
3

>> ForAll(GetProperties(ParseJSON("{""a"":1,""b"":{}}").b),{Name:ThisRecord.Name,StrValue:Text(ThisRecord.Value)})
