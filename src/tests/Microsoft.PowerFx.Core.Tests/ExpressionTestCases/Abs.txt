

// ******** INTEGER PARAMETERS ********
// Positive Integer Parameters

>> Abs(0)
0

>> Abs(2)
2

>> Abs(2147483647) // testing a large postive number (+/- 2^31) the upper limit for an integer-type variable 
2147483647

>> Abs(9007199254740992) // testing a really large positive number (+/- 2^53-1) the upper limit for a long-type variable
9007199254740992

// Negative Integer Parameters

>> Abs(-0)
0

>> Abs(-2)
2

>> Abs(-2147483647) // testing a large negative number (+/- 2^31) the lower limit for an integer-type variable 
2147483647

>> Abs(-9007199254740992) // testing a really large negative number (+/- 2^53) the lower limit for a long-type variable
9007199254740992

// ******** DECIMAL PARAMETERS ********
>> Abs(2.2)
2.2

>> Abs(-2.2)
2.2

>> Abs(123456.789)
123456.789

>> Abs(-123456.789)
123456.789

// ******** NULL and ERROR PARAMETERS ********
>> Abs(1/0)
Error({Kind:ErrorKind.Div0})

//Excel returns 0
>> Abs(Blank())
0

//Excel returns true
>> Abs(Blank()) = 0
true

>> IsBlank(Abs(Blank()))
false

// ******** STRING PARAMETERS ********

>> Abs("sampleText")
Error({Kind:ErrorKind.InvalidArgument})

>> Abs("10")
10

>> Abs("-10.123")
10.123

// ******** BOOLEAN PARAMETERS ********

>> Abs(true)
1

>> Abs(false)
0

>> Abs("true")
Error({Kind:ErrorKind.InvalidArgument})

>> Abs("false")
Error({Kind:ErrorKind.InvalidArgument})

// ******** Date values and functions with dates ********
>> Date(1976, 12,23) - Date(2000, 01, 01)
-8409

>> Abs(Date(1976, 12,23) - Date(2000, 01, 01))
8409

>> Abs(Date(2022, 10, 1))
44835
