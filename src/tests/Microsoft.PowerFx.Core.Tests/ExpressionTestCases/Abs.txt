

// ******** INTEGER PARAMETERS ********
// Positive Integer Parameters

>> Abs(0)
0

>> Abs(2)
2

>> Abs(2147483649) // testing a number larger than the interger range 
2147483649

>> Abs(9007199254740991) // testing a number larger than the float range (-9007199254740991 to 9007199254740991, (+/- 2^53 - 1))
9007199254740991

// Negative Integer Parameters

>> Abs(-0)
0

>> Abs(-2)
2

>> Abs(-2147483649) // testing a number smaller than the interger range 
2147483649

>> Abs(-9007199254740991) // testing a number smaller than the float range (-9007199254740991 to 9007199254740991, (+/- 2^53 - 1))
9007199254740991

// ******** DECIMAL PARAMETERS ********
>> Abs(2.2)
2.2

>> Abs(-2.2)
2.2

>> Abs(123456.789)
123456.789

>> Abs(-123456.789)
123456.789

// ******** NULL and ERROR PARAMETERS ********
>> Abs(1/0)
Error({Kind:ErrorKind.Div0})

//Excel returns 0
>> Abs(Blank())
0

//Excel returns true
>> Abs(Blank()) = 0
true

>> IsBlank(Abs(Blank()))
false

// ******** STRING PARAMETERS ********

>> Abs("sampleText")
Error({Kind:ErrorKind.InvalidArgument})

>> Abs("10")
10

>> Abs("-10.123")
10.123

// ******** BOOLEAN PARAMETERS ********

>> Abs(true)
1

>> Abs(false)
0

>> Abs("true")
Error({Kind:ErrorKind.InvalidArgument})

>> Abs("false")
Error({Kind:ErrorKind.InvalidArgument})

// ******** Date values and functions with dates ********
>> Date(1976, 12,23) - Date(2000, 01, 01)
-8409

>> Abs(Date(1976, 12,23) - Date(2000, 01, 01))
8409

>> Abs(Date(2022, 10, 1))
44835
