// ROUNDUP(NUMBER, NUMBER_OF_DIGITS)

// ******** INTEGER PARAMETERS ********

// Positive "NUMBER" and "NUMBER_OF_DIGITS" parameter

>> RoundUp(0,0)
0

>> RoundUp(0,1)
0

>> RoundUp(230,0)
230

>> RoundUp(230,1)
230

>> RoundUp(230,2)
230

>> RoundUp(230,3)
230


// Negative "NUMBER" Parameter

>> RoundUp(-0,1)
0

>> RoundUp(-456,0)
-456

>> RoundUp(-456,1)
-456

>> RoundUp(-456,2)
-456

>> RoundUp(-456,3)
-456


// Negative "NUMBER_OF_DIGITS" Parameter

>> RoundUp(3456,-1)
3460

>> RoundUp(3456,-2)
3500

>> RoundUp(3456,-3)
4000

>> RoundUp(3456,-4)
10000

// Negative "NUMBER" and "NUMBER_OF_DIGITS" parameter

>> RoundUp(-3456,-1)
-3460

>> RoundUp(-3456,-2)
-3500

>> RoundUp(-3456,-3)
-4000

>> RoundUp(-3456,-4)
-10000

// ******** NULL and ERROR PARAMETERS ********
>> RoundUp(1/0,1)
#Error(Kind=Div0)

>> RoundUp(1,1/0)
#Error(Kind=Div0)

>> RoundUp(Blank(),1)
0

>> RoundUp(Blank(), -1)
0

>> RoundUp(56.78, Blank())
57

>> RoundUp(-56.78, Blank())

>> RoundUp(Blank(),Blank())
0

// ******** STRING PARAMETERS ********

>> RoundUp("sampleText",1)
#Error(Kind=InvalidArgument)

>> RoundUp("10.5678","ABC")

>> RoundUp("-10.5678","2")

>> RoundUp("-10.5678","-2")

>> RoundUp("-10.123", "1.234")
